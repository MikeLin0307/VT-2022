````bash
#!/bin/bash
valid=true
count=1
((result=0))

while [ $valid ]
do
echo $count
((result = result + count))

if [ $count -eq 10 ];
then
break
fi
((count++))

done

echo '1+2+....+10 = '$result
````

````bash
#!/bin/bash

((result = 0 ))

for (( counter=i; counter<=10; counter++ ))
do
echo -n "$counter "
((result = result + counter))
done

echo -n ", sum = "$result
printf "\n"
````
